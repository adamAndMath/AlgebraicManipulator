assume Div(Complex a)
    lim<x>(a,div(x,x))=1;

assume Resolve(Complex a, Func<Complex,Complex> f)
    lim<x>(a,eval(f,x))=eval(f,x);

assume Func(Complex a,Func<Complex,Complex> f,Func<List<Complex,Complex>,Complex> g)
    lim<x>(a,eval(f,eval(g(x))))=eval(f,lim<x>(a,eval(g,x)))=lim<x>(eval(g,a),eval(f,x));

assume Reciprocal(Complex a)
    lim<x>(\inf,div(1,x))=0;
